```SQL
SELECT  CONCAT('/home/grep/src/',CONCAT(A.BOARD_ID,CONCAT('/',CONCAT(CONCAT(B.file_id,B.FILE_NAME ),FILE_EXT)))) AS FILE_PATH
-- CONCAT 열심히 썼습니다.
FROM USED_GOODS_BOARD A JOIN USED_GOODS_FILE B ON A.BOARD_ID=B.BOARD_ID
WHERE A.VIEWS=
(SELECT MAX(VIEWS)
FROM USED_GOODS_BOARD )
ORDER BY B.FILE_ID DESC
```
