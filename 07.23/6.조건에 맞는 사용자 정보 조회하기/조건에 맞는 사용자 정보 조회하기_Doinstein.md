```SQL
SELECT U.USER_ID, U.NICKNAME, U.CITY||' '||U.STREET_ADDRESS1||' '||U.STREET_ADDRESS2 AS "전체주소", SUBSTR(TLNO,1,3)||'-'||SUBSTR(TLNO,4,4)||'-'||SUBSTR(TLNO,8,4) AS "전화번호"
    FROM (SELECT WRITER_ID, COUNT(WRITER_ID) AS CNT
          FROM USED_GOODS_BOARD
          GROUP BY WRITER_ID) B
    JOIN USED_GOODS_USER U
    ON B.WRITER_ID = U.USER_ID
    WHERE B.CNT >=3
    ORDER BY USER_ID DESC;
```
