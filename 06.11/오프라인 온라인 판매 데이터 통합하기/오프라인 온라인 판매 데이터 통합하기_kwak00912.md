오프라인/온라인 판매 데이터 통합하기_kwak00912

```sql
SELECT *
FROM 
-- 초기에는 A, B로 테이블을 지정했었는데, 
-- 이상해서 다시 찾아보니 굳이 지정할 필요가 없었음
(SELECT  
TO_CHAR(SALES_DATE,'YYYY-MM-DD') AS SALES_DATE,
PRODUCT_ID,
USER_ID,
SALES_AMOUNT
FROM ONLINE_SALE
WHERE TO_CHAR(SALES_DATE,'MM') = '03'

 -- UNION / UNION ALL 차이
-- UNION 중복되는 행을 하나만 표시
-- UNION ALL 중복 제거 없이 모두 합쳐서 보여줌
UNION ALL

SELECT
TO_CHAR(SALES_DATE,'YYYY-MM-DD') AS SALES_DATE,
PRODUCT_ID,
-- OFFLINE_SALE 테이블의 판매 데이터의 USER_ID 값은 NULL 로 표시
-- 즉 NULL 값만 있는 열을 USER_ID로 지정하면 됨(OFFLINE_SALE 테이블엔 USER_ID 열이 없음)
NULL AS USER_ID,
SALES_AMOUNT
FROM  OFFLINE_SALE
WHERE TO_CHAR(SALES_DATE,'MM') = '03'
)

ORDER BY SALES_DATE ASC, PRODUCT_ID ASC, USER_ID ASC;
```
