```SQL
SELECT CAR_TYPE, COUNT(CAR_TYPE) AS CARS
FROM CAR_RENTAL_COMPANY_CAR 
WHERE OPTIONS LIKE '%통풍시트%' OR
      OPTIONS LIKE '%열선시트%' OR
      OPTIONS LIKE '%가죽시트%' -- LIKE와 IN은 동시에 사용 불가 > 정규식 활용
GROUP BY CAR_TYPE
ORDER BY CAR_TYPE;
```
