```SQL
SELECT FOOD_TYPE, REST_ID, REST_NAME, FAVORITES
FROM REST_INFO
--WHERE FAVORITES=ANY(SELECT MAX(FAVORITES) FROM REST_INFO GROUP BY FOOD_TYPE)
--첫번째 접근 MAX값을 뽑아내서 같은애들을 찾는다=> 큰오류 발생 만약 MAX값이 아닌 애들중에
--다른 그룹의 MAX값이랑 같으면 출력되는 오류 발생...
WHERE (FOOD_TYPE,FAVORITES) IN
(SELECT FOOD_TYPE, MAX(FAVORITES) FROM REST_INFO GROUP BY FOOD_TYPE)
--그래서 최대값을 가진 즐겨찾기+ 푸드타입까지 같은 것을 찾는 (FOOD_TYPE,FAVORITES) IN 을 활용
--해결했다!
GROUP BY FOOD_TYPE, REST_ID, REST_NAME, FAVORITES
ORDER BY FOOD_TYPE DESC
```
