```
SELECT B.TITLE, R.BOARD_ID, R.REPLY_ID, R.WRITER_ID,
        R.CONTENTS, TO_CHAR(R.CREATED_DATE, 'YYYY-MM-DD') AS CREATED_DATE
    FROM USED_GOODS_BOARD B
    JOIN USED_GOODS_REPLY R
    ON B.BOARD_ID = R.BOARD_ID -- BOARD_ID 컬럼으로 조인
    WHERE TO_CHAR(B.CREATED_DATE,'YYYYMM') = '202210' -- 2022년10월 조건
    ORDER BY R.CREATED_DATE, B.TITLE; 
```
