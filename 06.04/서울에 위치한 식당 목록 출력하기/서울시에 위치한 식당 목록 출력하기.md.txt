SELECT DISTINCT(I.REST_ID), I.REST_NAME, I.FOOD_TYPE, I.FAVORITES, I.ADDRESS, ROUND(AVG(R.REVIEW_SCORE), 2) AS SCORE  
FROM REST_INFO I JOIN REST_REVIEW R ON I.REST_ID = R.REST_ID 
WHERE SUBSTR(I.ADDRESS, 1, 2) = '서울'
GROUP BY I.REST_ID, I.REST_NAME, I.FOOD_TYPE, I.FAVORITES, I.ADDRESS
ORDER BY SCORE DESC, I.FAVORITES DESC

-- 처음에 중복 데이터를 제거해야 한다는 생각을 못해서 해멨고, 마지막에 GROUP BY를 해줘야 ROUND(AVG(R.REVIEW_SCORE))가 실행된다는 것을 깨달았다.. 
-- 그리고 서울특별시만 있는줄 알았는데 서울시가 있어서 이것도 한참 해메게 했다..
