```sql
-- oracle에서 DATE type은 반드시 TO_CHAR
SELECT MEMBER_ID, MEMBER_NAME, GENDER, TO_CHAR(DATE_OF_BIRTH,'YYYY-MM-DD') AS DATE_OF_BIRTH
FROM MEMBER_PROFILE
-- EXTRACT(MONTH FROM COLUMN)이 형태가 난 편하다
-- IS NOT NULL 조건 빼먹지 말기
WHERE EXTRACT(MONTH FROM DATE_OF_BIRTH) = '03' AND
TLNO IS NOT NULL AND GENDER = 'W'
-- 정렬 조건은 문제 읽으면서 다 쓰기
ORDER BY MEMBER_ID;

# 여성 조건 빼먹음 ㅠ
SELECT MEMBER_ID, MEMBER_NAME, GENDER, TO_CHAR(DATE_OF_BIRTH,'YYYY-MM-DD') AS DATE_OF_BIRTH
FROM MEMBER_PROFILE
WHERE EXTRACT(MONTH FROM DATE_OF_BIRTH) = '03' AND
TLNO IS NOT NULL
ORDER BY MEMBER_ID;
```
