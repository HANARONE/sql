```sql
SELECT BOOK_ID,TO_CHAR(PUBLISHED_DATE,'YYYY-MM-DD') from BOOK 
-- 원래 PUBLISHED_DATE는 시간까지 나와있어서 날짜까지만 나오게 하기위해 TO_CAHR사용
WHERE TO_CHAR(PUBLISHED_DATE,'YYYY') = '2021' 
--여기서 그냥 PUBLISHED_DATE='2021' 하면 되는줄 알았는데
--여기도 TO_CHAR(PUBLISHED_DATE,'YYYY') = '2021'이렇게 해야함
AND CATEGORY='인문' 
ORDER BY PUBLISHED_DATE ASC;


```
