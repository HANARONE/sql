저자 별 카테고리 별 매출액 집계하기_KWAK

```SQL
-- 3개 테이블을 합친다는 것만 주의
-- 어떤 순서로 합칠지만 잘 생각해서 풀면 될 듯
SELECT B.AUTHOR_ID, C.AUTHOR_NAME, B.CATEGORY, SUM(B.PRICE*A.SALES) AS TOTAL_SALES
FROM 
BOOK_SALES A LEFT JOIN BOOK B
ON A.BOOK_ID = B.BOOK_ID
LEFT JOIN AUTHOR C
ON B.AUTHOR_ID = C.AUTHOR_ID
WHERE TO_CHAR(A.SALES_DATE, 'YYYY-MM') = '2022-01' 
GROUP BY B.AUTHOR_ID, C.AUTHOR_NAME, B.CATEGORY
ORDER BY 1, 3 DESC
;
```
