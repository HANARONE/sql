```SQL
SELECT A.AUTHOR_ID, B.AUTHOR_NAME, A.CATEGORY, SUM(C.SALES * A.PRICE) AS TOTAL_SALES
-- 원하는 셀들을 가지고 옵니다. SUM(C.SALES * A.PRICE) 이게 이문제의 포인트 같습니다. 
FROM BOOK A, AUTHOR B, BOOK_SALES C
WHERE A.AUTHOR_ID = B.AUTHOR_ID AND A.BOOK_ID = C.BOOK_ID AND
TO_CHAR(C.SALES_DATE,'YYYY-MM') = '2022-01'
-- TO_CHAR은 이제 쉽게 하고요 사실 3가지 테이블을 조인하는건 제기억에 처음이었는데 그냥 쉽게 했습니다.
GROUP BY A.AUTHOR_ID, B.AUTHOR_NAME, A.CATEGORY
-- 그룹화는 해도해도 헷갈리는데 SUM같은 함수 빼고 다 써주면 되는것 같습니다.
ORDER BY A.AUTHOR_ID ASC, A.CATEGORY DESC;
```
