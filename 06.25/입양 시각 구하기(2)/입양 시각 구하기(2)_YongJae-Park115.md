```sql
SELECT
    A.HOUR,
    (CASE WHEN B.COUNT IS NULL THEN 0
    ELSE B.COUNT END) COUNT
FROM (
    SELECT LEVEL-1 AS HOUR
    FROM DUAL
    CONNECT BY LEVEL < 25
    ) A
LEFT JOIN(
    SELECT TO_CHAR(DATETIME, 'hh24') HOUR, COUNT(*) AS COUNT
    FROM ANIMAL_OUTS
    GROUP BY TO_CHAR(DATETIME, 'hh24')
    ) B
ON A.HOUR = B.HOUR
ORDER BY A.HOUR;
-- LEVEL 함수 알아야 풀 수 있는듯
-- CONNECT BY로 LEVEL 범위 설정
```
