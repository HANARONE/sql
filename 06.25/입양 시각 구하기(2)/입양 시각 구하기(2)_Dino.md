```sql
-- ALIAS 지정
SELECT B.HOUR, COUNT(A.HOUR) AS COUNT
-- 계층 데이터를 위해 CONNECT BY 절
FROM (SELECT TO_CHAR(DATETIME, 'HH24') HOUR FROM ANIMAL_OUTS) A,
(SELECT LEVEL-1 HOUR FROM DUAL CONNECT BY LEVEL <= 24) B
-- OUTER JOIN 연산자 (+) 를 조인시킬 값 없는 쪽에
WHERE B.HOUR = A.HOUR(+)
GROUP BY B.HOUR
ORDER BY B.HOUR;
```
