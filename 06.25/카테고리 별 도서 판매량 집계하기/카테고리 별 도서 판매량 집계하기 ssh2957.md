```SQL
SELECT CATEGORY, SUM(B.SALES) AS TOTAL_SALES
-- 문제에서 원하는 애들 셀렉하고 그중 SALES의 총합이니깐 SUM을 해준다 아무생각안하고
-- 숫자세는거니깐 COUNT쓰면 틀림 
FROM BOOK A, BOOK_SALES B
WHERE A.BOOK_ID= B.BOOK_ID AND TO_CHAR(B. SALES_DATE, 'YYYY-MM')='2022-01'
-- 조인해주고 '2022-01' 날짜 조건 해주고 풀면 끝 
GROUP BY CATEGORY
ORDER BY CATEGORY ASC
```
