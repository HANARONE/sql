```SQL
SELECT A.WRITER_ID, B. NICKNAME, SUM(A.PRICE) AS TOTAL_SALES
-- 평범한 그룹화 문제
FROM USED_GOODS_BOARD A, USED_GOODS_USER B
WHERE A.WRITER_ID = B.USER_ID AND STATUS='DONE'
-- 그룹화 해주고 거래가 완료된 중고거래의 총금액이니깐 STATUS='DONE' 를 걸어주고 찾는다.
GROUP BY A.WRITER_ID, B. NICKNAME
HAVING SUM(A.PRICE)>=700000 
-- 여기서 70만 이상인데 =빼서 한번 틀렸었음
-- 중요한건 SUM(A.PRICE)>=700000 조건은 그룹화에 걸려있는 조건이기 때문에 WHERE 이 아닌 HAVING 절에 걸어준다
ORDER BY SUM(A.PRICE)
```
