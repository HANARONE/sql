```sql
-- 정답
SELECT 
    TO_CHAR(SALES_DATE,'YYYY') AS YEAR, 
    TO_NUMBER(TO_CHAR(SALES_DATE,'MM')) AS MONTH, 
    GENDER, 
    COUNT(DISTINCT A.USER_ID) AS USERS
FROM USER_INFO A 
    JOIN ONLINE_SALE B 
    ON A.USER_ID = B.USER_ID
WHERE GENDER IS NOT NULL
GROUP BY TO_CHAR(SALES_DATE,'YYYY'), TO_NUMBER(TO_CHAR(SALES_DATE,'MM')), GENDER
ORDER BY YEAR, MONTH, GENDER

-- 내 기존 제출코드
SELECT 
    TO_CHAR(SALES_DATE,'YYYY') AS YEAR, 
    TO_NUMBER(TO_CHAR(SALES_DATE,'MM')) AS MONTH, 
    GENDER, 
    COUNT(DISTINCT A.USER_ID) AS USERS
FROM USER_INFO A 
    LEFT JOIN ONLINE_SALE B 
    ON A.USER_ID = B.USER_ID
WHERE GENDER IS NOT NULL
GROUP BY TO_CHAR(SALES_DATE,'YYYY'), TO_NUMBER(TO_CHAR(SALES_DATE,'MM')), GENDER
ORDER BY YEAR, MONTH, GENDER

-- 내 기존 제출코드와 비슷한 정답
SELECT
    TO_CHAR(SALES_DATE, 'YYYY') YEAR, 
    TO_NUMBER(TO_CHAR(SALES_DATE, 'MM')) MONTH, 
    GENDER, 
    COUNT(DISTINCT U.USER_ID) USERS
FROM ONLINE_SALE O
    LEFT OUTER JOIN USER_INFO U
    ON O.USER_ID = U.USER_ID
WHERE GENDER IS NOT NULL
GROUP BY TO_CHAR(SALES_DATE, 'YYYY'), TO_CHAR(SALES_DATE, 'MM'), GENDER
ORDER BY 1, 2, 3;
-- 2번째것과 3번째것 차이를 모르겠음
```
