```sql
-- MONTH 처리 주의, DISTICT 중복처리
SELECT TO_CHAR(SALES_DATE, 'YYYY') YEAR, TO_NUMBER(TO_CHAR(SALES_DATE, 'MM')) MONTH, GENDER, COUNT(DISTINCT I.USER_ID) USERS
FROM ONLINE_SALE O
JOIN USER_INFO I
ON O.USER_ID = I.USER_ID
-- NULL 값 제외 (문제 조건)
WHERE GENDER IS NOT NULL
-- 년,월,성별 별로 grouping
GROUP BY TO_CHAR(SALES_DATE, 'YYYY'), TO_CHAR(SALES_DATE, 'MM'), GENDER
-- 정렬조건
ORDER BY 1, 2, 3;
```
