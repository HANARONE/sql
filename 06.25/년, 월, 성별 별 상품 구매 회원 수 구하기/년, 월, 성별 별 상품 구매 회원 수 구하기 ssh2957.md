```SQL
SELECT TO_CHAR(B.SALES_DATE, 'YYYY') AS YEAR, TO_NUMBER(TO_CHAR(B.SALES_DATE, 'MM')) AS MONTH, A.GENDER
,	COUNT(DISTINCT B.USER_ID) AS USERS
-- 이문제에서는 MM형태를 어떻게 1,2,3 으로 보이냐를 몰랐는데 TO_NUMBER을 통해 01이 아닌 정수 1로 변환시켜줌
-- 또한 USER_ID에 중복이 들어가면 안되기 때문에 DISTINCT를 사용했다.
       
FROM USER_INFO A
JOIN ONLINE_SALE B ON A.USER_ID = B.USER_ID
WHERE A.GENDER IS NOT NULL
-- GENDER가 NULL일때는 나오지 않게 하기위해서 IS NOT NULL 사용
GROUP BY TO_CHAR(B.SALES_DATE, 'YYYY'), TO_NUMBER(TO_CHAR(B.SALES_DATE, 'MM')), A.GENDER
-- 셀렉문에 정의된 순서대로 정렬할때 1,2,3 사용해도 된다함
ORDER BY 1,2,3 ; 

```
