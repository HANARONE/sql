```sql
SELECT DISTINCT(CAR_ID) AS CAR_ID
        ,CASE WHEN CAR_ID IN (SELECT CAR_ID FROM CAR_RENTAL_COMPANY_RENTAL_HISTORY
                             WHERE TO_CHAR(START_DATE,'YYYYMMDD') <= '20221016'
                                    AND TO_CHAR(END_DATE,'YYYYMMDD') >= '20221016')
              THEN '대여중' ELSE '대여 가능' END AS AVAILABILITY
FROM CAR_RENTAL_COMPANY_RENTAL_HISTORY
ORDER BY CAR_ID DESC;
```
