```SQL
  SELECT TO_NUMBER(TO_CHAR(START_DATE, 'MM')) AS MONTH, CAR_ID, COUNT(HISTORY_ID) AS RECORDS
  FROM CAR_RENTAL_COMPANY_RENTAL_HISTORY
  WHERE TO_CHAR(START_DATE, 'YYYY-MM') IN ('2022-08', '2022-09', '2022-10')
    AND TO_CHAR(END_DATE, 'YYYY-MM') IN ('2022-08', '2022-09', '2022-10')
  GROUP BY TO_CHAR(START_DATE, 'MM'), CAR_ID
  HAVING COUNT(HISTORY_ID) >= 5
ORDER BY 3,2
-- 총 레코드 수가 5이상인 CAR ID 까지는 뽑아냈는데 그후에 달별로 나누는 법을 모르겠습니다...

```
