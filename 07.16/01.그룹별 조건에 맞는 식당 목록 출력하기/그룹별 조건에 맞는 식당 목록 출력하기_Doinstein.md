```SQL
SELECT P.MEMBER_NAME, R.REVIEW_TEXT, TO_CHAR(R.REVIEW_DATE,'YYYY-MM-DD') REVIEW_DATE
FROM MEMBER_PROFILE P ,
     REST_REVIEW R,
     (SELECT MEMBER_ID, RANK() OVER(ORDER BY COUNT(MEMBER_ID) DESC) AS RANK
       FROM REST_REVIEW
       GROUP BY MEMBER_ID) S
WHERE P.MEMBER_ID = R.MEMBER_ID AND
      P.MEMBER_ID = S.MEMBER_ID AND
      S.RANK = 1
ORDER BY review_date, review_text
```
